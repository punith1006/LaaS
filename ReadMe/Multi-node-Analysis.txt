private async selectComputeNode(requiredVcpu: number, requiredMemoryMb: number, requiredVramMb: number): Promise<Node> {
  const healthyNodes = await this.prisma.node.findMany({
    where: { status: 'healthy' }
  });
  
  if (healthyNodes.length === 0) {
    throw new ServiceUnavailableException('No healthy compute nodes available');
  }
  
  // Score each node: (availableCPU * 0.3) + (availableRAM * 0.4) + (availableVRAM * 0.3)
  const scoredNodes = await Promise.all(
    healthyNodes.map(async (node) => {
      const activeSessions = await this.prisma.session.findMany({
        where: { nodeId: node.id, status: { in: RESOURCE_CONSUMING_STATUSES as any[] } },
        include: { computeConfig: true }
      });
      
      const usedResources = {
        vcpu: activeSessions.reduce((sum, s) => sum + (s.allocatedVcpu ?? s.computeConfig.vcpu), 0),
        memoryMb: activeSessions.reduce((sum, s) => sum + (s.allocatedMemoryMb ?? s.computeConfig.memoryMb), 0),
        vramMb: activeSessions.reduce((sum, s) => sum + (s.allocatedGpuVramMb ?? s.computeConfig.gpuVramMb), 0),
      };
      
      const metadata = node.metadata as Record<string, unknown> | null;
      const allocatable = {
        vcpu: (metadata?.allocatableVcpu as number) ?? node.totalVcpu - 2,
        memoryMb: (metadata?.allocatableMemoryMb as number) ?? node.totalMemoryMb - 10240,
        vramMb: (metadata?.allocatableGpuVramMb as number) ?? node.totalGpuVramMb - 1024,
      };
      
      const available = {
        vcpu: allocatable.vcpu - usedResources.vcpu,
        memoryMb: allocatable.memoryMb - usedResources.memoryMb,
        vramMb: allocatable.vramMb - usedResources.vramMb,
      };
      
      const score = (available.vcpu * 0.3) + (available.memoryMb * 0.4) + (available.vramMb * 0.3);
      
      return { node, available, score };
    })
  );
  
  scoredNodes.sort((a, b) => b.score - a.score);
  const selectedNode = scoredNodes[0];
  
  if (selectedNode.available.vcpu < requiredVcpu || 
      selectedNode.available.memoryMb < requiredMemoryMb || 
      selectedNode.available.vramMb < requiredVramMb) {
    throw new ConflictException('Insufficient resources on all nodes');
  }
  
  return selectedNode.node;
}

def mount_remote_storage(storage_node_ip: str, storage_uid: str, mount_path: str) -> Optional[str]:
    """
    Mount user storage from a remote node via NFS.
    Returns None on success, or error string on failure.
    """
    if not storage_uid or not STORAGE_UID_PATTERN.match(storage_uid):
        return f"Invalid storage_uid: {storage_uid}"
    
    # Ensure mountpoint exists
    os.makedirs(mount_path, exist_ok=True)
    
    # Check if already mounted
    result = subprocess.run(['mount'], capture_output=True, text=True)
    mounts = result.stdout
    if f" {mount_path} " in f" {mounts} ":
        logger.info(f"Storage already mounted at {mount_path}")
        return None
    
    # Mount via NFS4
    nfs_source = f"{storage_node_ip}:/datapool/users/{storage_uid}"
    mount_opts = "rw,sync,proto=tcp,vers=4.1,rsize=131072,wsize=131072,hard,intr"
    
    try:
        result = subprocess.run(
            ["mount", "-t", "nfs4", "-o", mount_opts, nfs_source, mount_path],
            capture_output=True, text=True, timeout=20
        )
        if result.returncode != 0:
            return f"NFS mount failed: {result.stderr.strip()}"
        
        logger.info(f"Mounted {nfs_source} at {mount_path}")
        return None
    except subprocess.TimeoutExpired:
        return "NFS mount timed out (>20s)"
    except Exception as e:
        return f"NFS mount error: {str(e)}"

The Problem

Node 2 later needs to launch the same user's container

def commit_container_state_with_export(container_name: str, storage_uid: Optional[str]) -> bool:
    if not storage_uid:
        return False

    user_image = f"laas-user-{storage_uid}:latest"
    
    try:
        # 1. Commit to local Docker
        logger.info(f"Committing {container_name} → {user_image}...")
        result = subprocess.run(
            ["docker", "commit", "--change", "LABEL laas.committed=true",
             "--change", f"LABEL laas.storage_uid={storage_uid}",
             container_name, user_image],
            capture_output=True, text=True, timeout=120
        )
        if result.returncode != 0:
            logger.error(f"Docker commit failed: {result.stderr}")
            return False
        
        # 2. Export to user's NFS-mounted storage
        # Mount user's storage first if not already mounted
        mount_path = f"{NFS_MOUNT_ROOT}/{storage_uid}"
        if not os.path.ismount(mount_path):
            # This would only happen if remote node; handle gracefully
            logger.warning(f"User storage not mounted at {mount_path}, skipping export")
            return True  # Commit succeeded locally; export is best-effort
        
        # 3. Create .system directory for Docker images
        system_dir = os.path.join(mount_path, '.system')
        os.makedirs(system_dir, exist_ok=True)
        
        # 4. Export image to tar file
        image_tar_path = os.path.join(system_dir, 'container-image.tar')
        logger.info(f"Exporting {user_image} → {image_tar_path}...")
        
        result = subprocess.run(
            ["docker", "save", "-o", image_tar_path, user_image],
            capture_output=True, text=True, timeout=600  # 10 min timeout for large images
        )
        if result.returncode != 0:
            logger.error(f"Docker export failed: {result.stderr}")
            return False
        
        logger.info(f"Successfully exported {user_image} to {image_tar_path}")
        return True

        logger.error(f"Docker commit/export timed out for {container_name}")

I. Docker Commit Cross-Node Problem — Detailed Analysis
Scenario: User "Alice" Multi-Node Experience
Session 1: Alice launches on Node 1 (default)
Backend selects Node 1 as compute node (only node available)
Alice runs sudo apt install pytorch → layer written to ephemeral writable layer
Alice saves ~/project/model.pth → lands on NFS mount (persistent)
Alice stops session
Orchestrator commits: docker commit laas-alice-session-1 laas-user-<alice_uid>:latest
Image saved to Node 1's local Docker daemon
Session 2: Alice launches on Node 2 (system selected for better headroom)
Backend selects Node 2 (scores higher due to available resources)
Orchestrator on Node 2 queries docker image inspect laas-user-<alice_uid>:latest
Image NOT found (it's on Node 1, not Node 2's daemon)
Falls back to base Selkies image
Container starts from scratch
Alice loses: All apt packages, pip packages, system configs, /etc changes
Alice keeps: ~/project/model.pth (it's on NFS)
Result: User experience broken — appears to be factory reset
Why This is Critical
The value proposition of LaaS is persistent stateful desktops. Users expect:
Packages they installed to survive restarts
System configurations to persist
Development environments to remain intact
Without solving the cross-node image problem, multi-node architecture breaks this promise.
